			else
			end

		end,
		["Admit"] = function ( ... )
			-- body
			isOn = gameObject:GetComponent("Toggle").isOn
		end,
		["ReturnButton"] = function ( ... )
			-- body
			_Choose_UI:SetActive(false)
			_Main_UI:SetActive(true)
		end,
		["PlayerLogo"] = function ( ... )
			-- body
			_Player_Message_UI:SetActive(true)
		end,
		["DiamondPlus"] = function ( ... )
			-- body
			print("DiamondPlusShow")
		end,
		["BeanPlus"] = function ( ... )
			-- body
			print("BeanPlusShow")
		end,
		["TicketPlus"] = function ( ... )
			-- body
			print("TicketPlusShow")
		end,
		["GoldZone"] = function ( ... )
			-- body
			print("GoldZone")
		end,
		["JoinRoom"] = function ( ... )
			-- body
			print("JoinRoom")
		end,
		["CreatRoom"] = function ( ... )
			-- body
			print("CreatRoom")
		end,
		["Store"] = function ( ... )
			-- body
		end,
		["Option"] = function ( ... )
			-- body
		end,
		["Mail"] = function ( ... )
			-- body
		end,

		end,

			-- body
		end,
		["Reward"] = function ( ... )
			-- body
		end,
		["More"] = function ( ... )
			-- body
		end,
		["ClosePlayerMessagePanel"] = function ( ... )
			-- body
			_Player_Message_UI:SetActive(false)
